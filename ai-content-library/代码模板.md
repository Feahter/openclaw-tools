### 001 批量重命名文件

```python
import os
import re

def rename_files(directory, pattern, replacement):
    """
    批量重命名文件
    """
    for filename in os.listdir(directory):
        if pattern in filename:
            new_filename = filename.replace(pattern, replacement)
            old_path = os.path.join(directory, filename)
            new_path = os.path.join(directory, new_filename)
            os.rename(old_path, new_path)
            print(f"Renamed: {filename} -> {new_filename}")

# 使用示例
rename_files("/path/to/files", "old_", "new_")
```

### 002 批量下载图片

```python
import requests
import os
from urllib.parse import urljoin

def download_images(urls, save_dir):
    """
    批量下载图片
    """
    os.makedirs(save_dir, exist_ok=True)
    
    for i, url in enumerate(urls):
        try:
            response = requests.get(url, timeout=10)
            filename = f"image_{i+1}.jpg"
            filepath = os.path.join(save_dir, filename)
            
            with open(filepath, 'wb') as f:
                f.write(response.content)
            print(f"Downloaded: {filename}")
        except Exception as e:
            print(f"Error: {e}")

# 使用示例
urls = [
    "https://example.com/img1.jpg",
    "https://example.com/img2.jpg"
]
download_images(urls, "./images")
```

### 003 自动发送邮件

```python
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart

def send_email(smtp_server, smtp_port, sender, password, receivers, subject, body):
    """
    发送邮件
    """
    msg = MIMEMultipart()
    msg['From'] = sender
    msg['To'] = ', '.join(receivers)
    msg['Subject'] = subject
    
    msg.attach(MIMEText(body, 'html'))
    
    try:
        server = smtplib.SMTP(smtp_server, smtp_port)
        server.starttls()
        server.login(sender, password)
        server.sendmail(sender, receivers, msg.as_string())
        print("Email sent successfully!")
    except Exception as e:
        print(f"Error: {e}")

# 使用示例
send_email(
    smtp_server="smtp.gmail.com",
    smtp_port=587,
    sender="your_email@gmail.com",
    password="your_password",
    receivers=["receiver@example.com"],
    subject="Test Email",
    body="<h1>Hello World</h1>"
)
```

### 001 批量下载文件

```javascript
const fs = require('fs');
const path = require('path');
const https = require('https');

async function downloadFiles(urls, saveDir) {
    if (!fs.existsSync(saveDir)) {
        fs.mkdirSync(saveDir, { recursive: true });
    }
    
    for (let i = 0; i < urls.length; i++) {
        const url = urls[i];
        const filename = `file_${i + 1}${path.extname(url)}`;
        const filepath = path.join(saveDir, filename);
        
        await new Promise((resolve, reject) => {
            https.get(url, (response) => {
                const file = fs.createWriteStream(filepath);
                response.pipe(file);
                file.on('finish', () => {
                    file.close();
                    console.log(`Downloaded: ${filename}`);
                    resolve();
                });
            }).on('error', (err) => {
                console.error(`Error: ${err.message}`);
                resolve();
            });
        });
    }
}

// 使用示例
const urls = [
    'https://example.com/file1.jpg',
    'https://example.com/file2.jpg'
];
downloadFiles(urls, './downloads');
```

### 002 API服务器

```javascript
const express = require('express');
const app = express();
const PORT = 3000;

app.use(express.json());

// 模拟数据
const data = [
    { id: 1, name: 'Item 1', price: 100 },
    { id: 2, name: 'Item 2', price: 200 },
    { id: 3, name: 'Item 3', price: 300 }
];

// GET 获取列表
app.get('/api/items', (req, res) => {
    res.json(data);
});

// GET 获取单个
app.get('/api/items/:id', (req, res) => {
    const item = data.find(i => i.id === parseInt(req.params.id));
    if (item) {
        res.json(item);
    } else {
        res.status(404).json({ error: 'Not found' });
    }
});

// POST 创建
app.post('/api/items', (req, res) => {
    const newItem = {
        id: data.length + 1,
        ...req.body
    };
    data.push(newItem);
    res.status(201).json(newItem);
});

app.listen(PORT, () => {
    console.log(`Server running on port ${PORT}`);
});

# Python版本爬虫
import requests
from bs4 import BeautifulSoup

def scrape_article(url):
    """
    爬取文章内容
    """
    response = requests.get(url)
    soup = BeautifulSoup(response.text, 'html.parser')
    
    # 获取标题
    title = soup.find('h1').text
    
    # 获取正文
    content = []
    for p in soup.find_all('p'):
        content.append(p.text)
    
    return {
        'title': title,
        'content': '\n'.join(content)
    }

# 使用示例
article = scrape_article('https://example.com/article')
print(f"标题: {article['title']}")
print(f"内容: {article['content'][:200]}")
```

### 001 自动备份

```python
import os
import shutil
import datetime

def backup_files(source_dir, backup_dir):
    """
    备份文件
    """
    timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
    backup_path = os.path.join(backup_dir, f"backup_{timestamp}")
    
    os.makedirs(backup_path, exist_ok=True)
    
    for filename in os.listdir(source_dir):
        source_path = os.path.join(source_dir, filename)
        if os.path.isfile(source_path):
            dest_path = os.path.join(backup_path, filename)
            shutil.copy2(source_path, dest_path)
            print(f"Backed up: {filename}")
    
    print(f"Backup completed: {backup_path}")

# 使用示例
backup_files("./source", "./backups")
```

### 002 日志分析

```python
import re
from collections import Counter

def analyze_logs(log_file):
    """
    分析日志文件
    """
    errors = []
    warnings = []
    
    with open(log_file, 'r') as f:
        for line in f:
            if 'ERROR' in line:
                errors.append(line)
            elif 'WARNING' in line:
                warnings.append(line)
    
    print(f"=== 日志分析报告 ===")
    print(f"Error 数量: {len(errors)}")
    print(f"Warning 数量: {len(warnings)}")
    
    # 常见错误
    error_patterns = [re.search(r'Error: (.+)', e) for e in errors]
    error_counts = Counter([m.group(1) for m in error_patterns if m])
    
    print(f"\n=== 常见错误 TOP 10 ===")
    for error, count in error_counts.most_common(10):
        print(f"{error}: {count}")

# 使用示例
analyze_logs("app.log")
```
